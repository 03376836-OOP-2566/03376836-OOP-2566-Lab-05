### สิ่งที่พบในการทดลอง
- โปรแกรม Build และ Run ได้เพราะclassได้กำหนดสิทธิในการเข้าถึงแล้ว
- ผลลัพท์ที่ได้ คือ Type ของค่าที่เรากำหนด
  
p.name = Rambo  type = System.String

p.id = 1987     type = System.String

p.income = 2500 type = System.Int32
